import Foundation
import simd

struct Scene3D {
    var models: [Model]
    var strokes: [BrushStroke]
    var camera: Camera
    var lighting: Lighting
    var cadHistory = CADHistoryTree()
    var constraintManager = GeometryConstraintManager()
    
    struct Camera {
        var position: SIMD3<Float>
        var target: SIMD3<Float>
        var up: SIMD3<Float>
        var fov: Float
        var nearPlane: Float
        var farPlane: Float
        
        static var `default`: Camera {
            Camera(position: SIMD3<Float>(0, 0, 3),
                   target: SIMD3<Float>(0, 0, 0),
                   up: SIMD3<Float>(0, 1, 0),
                   fov: 45,
                   nearPlane: 0.1,
                   farPlane: 100)
        }
    }
    
    struct Lighting {
        var ambientColor: SIMD3<Float>
        var directionalLight: DirectionalLight
        
        struct DirectionalLight {
            var direction: SIMD3<Float>
            var color: SIMD3<Float>
            var intensity: Float
        }
        
        static var `default`: Lighting {
            Lighting(ambientColor: SIMD3<Float>(0.2, 0.2, 0.2),
                     directionalLight: DirectionalLight(direction: SIMD3<Float>(0, -1, -1),
                                                        color: SIMD3<Float>(1, 1, 1),
                                                        intensity: 0.8))
        }
    }
    
    init() {
        self.models = []
        self.strokes = []
        self.camera = .default
        self.lighting = .default
        
        // Configurar closures de constraint manager apuntando a este Scene3D
        setupConstraintClosures()
    }
    
    mutating func addModel(_ model: Model) {
        models.append(model)
        // Resolver constraints tras agregar modelo
        constraintManager.resolveConstraints()
    }
    
    mutating func addStroke(_ stroke: BrushStroke) {
        strokes.append(stroke)
        // Resolver constraints tras agregar stroke
        constraintManager.resolveConstraints()
    }
    
    /// Configura el solver de constraints con acceso a las entidades 3D de la escena
    private mutating func setupConstraintClosures() {
        // Nota: Como Scene3D es un struct, necesitamos capturar weak reference via proxy
        // Esta configuracion se hace en CanvasViewModel donde tenemos @Published
    }
    
    /// Configurar constraint manager con provider/updater (llamado desde CanvasViewModel)
    mutating func configureConstraintSolver() {
        let modelRef = models
        constraintManager.entityPositionProvider = { [weak modelRef] entityID in
            guard let models = modelRef else { return nil }
            for model in models {
                for vertex in model.meshes.flatMap({ $0.vertices }) {
                    // Nota: Los vertices necesitan IDs unicos para matching
                    // Por ahora retornamos la primera coincidencia basica
                    if vertex.id == entityID {
                        return vertex.position
                    }
                }
            }
            return nil
        }
        constraintManager.entityPositionUpdater = { [weak modelRef] entityID, newPosition in
            guard var models = modelRef else { return }
            for i in 0..<models.count {
                for j in 0..<models[i].meshes.count {
                    for k in 0..<models[i].meshes[j].vertices.count {
                        if models[i].meshes[j].vertices[k].id == entityID {
                            models[i].meshes[j].vertices[k].position = newPosition
                        }
                    }
                }
            }
        }
    }
}
